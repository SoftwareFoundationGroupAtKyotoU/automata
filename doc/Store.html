def transaction(readonly=nil, &block)
  return block.call(self) if @store

  begin
    @store = store_class.new(@path, true)
    @store.transaction((readonly.nil? && @readonly) || readonly) do
      block.call(self)
    end
  ensure
    @store = nil
  end
end
