def to_hash()
  hash = {
    'login'    => login,
    'token'    => token,
    'name'     => name,
    'ruby'     => ruby,
  }
  hash['report'] = {} unless report.empty?
  report.each{|k,v| hash['report'][k] = v.to_hash}
  return hash
end
