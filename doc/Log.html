def pop()
  popped_id = nil
  transaction do
    root = @store.roots.min_by{|r| oldest_id(r)}
    if root
      popped_id = oldest_id(root)
      @store.roots.each do |r|
        @store[r] = (@store[r]||[]).reject{|x| x['id'] == popped_id }
      end
    end
  end
  return popped_id
end

def size()
  transaction do

    return @store.roots.map{|r| (@store[r]||[]).size}.max
  end
end
